int findMax(vector<int> sticker, int start, int end) {
	int prevMaxIncluded = 0;
	int prevMaxNotIncluded = 0;

	for (int i = start; i <= end; i++) {
		int curMaxIncluded = sticker[i] + prevMaxNotIncluded;
		int curMaxNotIncluded = prevMaxIncluded;
		if (prevMaxIncluded < prevMaxNotIncluded) {
			curMaxNotIncluded = prevMaxNotIncluded;
		}
		prevMaxIncluded = curMaxIncluded;
		prevMaxNotIncluded = curMaxNotIncluded;
	}

	if (prevMaxNotIncluded < prevMaxIncluded) {
		return prevMaxIncluded;
	}
	else {
		return prevMaxNotIncluded;
	}
}
